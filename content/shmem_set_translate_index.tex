\apisummary{
    Returns the index of the \ac{PE} in the set.
}

\begin{apidefinition}
\begin{Csynopsis}
int @\FuncDecl{shmem\_set\_translate\_index}@(shmem_set_t *from_set, shmem_set_t *to_set, int from_index, int *to_index)
\end{Csynopsis}

\begin{apiarguments}
     \apiargument{IN}{from\_set}{A valid set. A valid set is either a pre-defined set or created by one the set creation operation.}
      \apiargument{IN}{to\_set}{A valid set. A valid set is either a pre-defined set or created by one the set creation operation.}
       \apiargument{IN}{from\_index}{A valid index in the from\_set}
      \apiargument{OUT}{to\_index}{A valid index in the to\_set}
\end{apiarguments}

\apidescription{
The shmem\_set\_translate\_index routine returns the index in the to\_set that corresponds to the \ac{PE} that has from\_index in from\_set. The value returned will be a valid non-negative integer if the \ac{PE} is part of the to\_set, else it will return a negative integer. }

\apireturnvalues{
shmem\_set\_translate\_index returns 0 upon success; otherwise, it returns a non-zero value.
}

\apinotes{ If the to\_set is SHMEM\_SET\_WORLD, it will return the \ac{PE} number as reported by shmem\_my\_pe.}

\end{apidefinition}
