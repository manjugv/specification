\apisummary{
    Returns the index of the \ac{PE} in the set.
}

\begin{apidefinition}
\begin{Csynopsis}
int @\FuncDecl{shmem\_set\_get\_index}@(shmem_set_t *set, int *index)
\end{Csynopsis}

\begin{apiarguments}
     \apiargument{IN}{set}{A valid set. A valid set is either a pre-defined set or created by one the set creation operation.}
     \apiargument{OUT}{index}{Index of the \ac{PE} in the set}
\end{apiarguments}

\apidescription{
The shmem\_set\_get\_index routine returns the index of the calling \ac{PE} in the set. The value returned will be a valid non-negative integer if the \ac{PE} is part of the set, else it will return a negative integer. }

\apireturnvalues{
shmem\_set\_get\_index returns 0 upon success; otherwise, it returns a non-zero value.
}

\apinotes{ None.}

\end{apidefinition}
